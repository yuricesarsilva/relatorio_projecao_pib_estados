eh_indice <- function(var) var %in% c("idx_vol_pib", "idx_deflator")

camada_base_100 <- function(var) {
  if (eh_indice(var)) {
    return(
      geom_hline(
        yintercept = 100,
        colour = "grey70",
        linewidth = 0.3,
        linetype = "dotted"
      )
    )
  }

  geom_blank()
}

formato_y <- function(var) {
  if (eh_indice(var))
    scale_y_continuous(labels = label_number(accuracy = 0.1, decimal.mark = ",",
                                             big.mark = "."))
  else
    scale_y_continuous(labels = label_number(big.mark = ".", decimal.mark = ",",
                                             accuracy = 1, scale = 1e-3,
                                             suffix = " bi"))
}

# ==============================================================================

# ==============================================================================

server <- function(input, output, session) {

  # ---- Autenticação -----------------------------------------------------------
  autenticado <- reactiveVal(FALSE)

  modal_login <- function(erro = FALSE) {
    modalDialog(
      title = tags$div(
        style = "text-align:center; font-size:1.1em; font-weight:600;",
        "Projeções do PIB Estadual — SEPLAN-RR"
      ),
      tags$div(
        style = "text-align:center; padding:4px 0 12px;",
        tags$p(
          style = "color:#666; font-size:0.9em; margin-bottom:20px;",
          "Painel em análise. Acesso restrito."
        ),
        passwordInput("senha_input", label = NULL,
                      placeholder = "Senha", width = "100%"),
        if (erro) tags$p(
          style = "color:#c0392b; font-size:0.85em; margin-top:6px;",
          "Senha incorreta. Tente novamente."
        )
      ),
      footer = tags$div(
        style = "text-align:center;",
        actionButton("btn_entrar", "Entrar",
                     class = "btn-primary",
                     style = "min-width:120px;")
      ),
      tags$script(HTML(
        "$(document).off('keydown.login').on('keydown.login', function(e){",
        "  if(e.key==='Enter'&&$('#senha_input').is(':visible')){$('#btn_entrar').click();}",
        "});"
      )),
      easyClose = FALSE,
      size = "s"
    )
  }

  observe({
    if (!autenticado()) showModal(modal_login())
  })

  observeEvent(input$btn_entrar, {
    if (identical(input$senha_input, SENHA_PAINEL)) {
      autenticado(TRUE)
      removeModal()
    } else {
      showModal(modal_login(erro = TRUE))
    }
  }, ignoreInit = TRUE)

  observeEvent(input$data_load_error, {
    showNotification(
      paste("Falha ao carregar os dados do painel:", input$data_load_error),
      type = "error",
      duration = NULL
    )
  })

  # Tema dia/noite
  observe({
    session$setCurrentTheme(
      if (isTRUE(input$modo_cor == "dark"))
        bs_theme(preset = "darkly", version = 5)
      else
        bs_theme(preset = "flatly", version = 5)
    )
  })

  # Dados: carregados via JS fetch -> input$*_raw -> read.csv(textConnection())
  serie <- reactive({
    req(input$serie_raw)
    read.csv(textConnection(input$serie_raw), na.strings = "")
  })
  vab_mac <- reactive({
    req(input$vab_mac_raw)
    read.csv(textConnection(input$vab_mac_raw), na.strings = "")
  })
  vab_ativ <- reactive({
    req(input$vab_ativ_raw)
    read.csv(textConnection(input$vab_ativ_raw), na.strings = "")
  })
  diagnostico_df <- reactive({
    req(input$diagnostico_raw)
    read.csv(textConnection(input$diagnostico_raw), na.strings = "")
  })

  ano_max <- reactive({
    n_proj <- input$n_proj
    if (is.null(n_proj) || is.na(n_proj)) {
      n_proj <- H_PAINEL
    }
    ANO_HIST_FIM + as.integer(n_proj)
  })

  # Dados filtrados
  dados_serie <- reactive({
    req(serie())
    serie() |>
      filter(geo == input$geo, variavel == input$variavel,
             ano <= ano_max()) |>
      mutate(tipo = factor(tipo, levels = c("Histórico", "Projetado")))
  })

  # ---- Aba 1: Série Histórica + Projeção ------------------------------------
  output$titulo_serie <- renderText({
    paste0(VARIAVEL_LABEL[input$variavel], " — ", input$geo)
  })

  output$plot_serie <- renderPlot({
      df <- dados_serie()

      shiny::validate(shiny::need(nrow(df) > 0, "Sem dados para a combinação selecionada."))

      unidade  <- if (eh_indice(input$variavel)) "Índice (2002 = 100)" else "R$ bilhões"

      ggplot(df, aes(x = ano)) +
        geom_ribbon(
          aes(ymin = lo95, ymax = hi95),
          data = \(d) filter(d, tipo == "Projetado", !is.na(lo95)),
          fill = COR_RIBBN, alpha = 0.20
        ) +
        vl +
        camada_base_100(input$variavel) +
        geom_line(aes(y = valor, colour = tipo), linewidth = 1.1) +
        geom_point(aes(y = valor, colour = tipo), size = 2.0) +
        scale_colour_manual(
          values = c(Histórico = COR_HIST, Projetado = COR_PROJ), name = ""
        ) +
        scale_x_continuous(breaks = seq(2002, ANO_PROJ_FIM, 3)) +
        formato_y(input$variavel) +
        labs(title    = paste0(VARIAVEL_LABEL[input$variavel], " — ", input$geo),
             subtitle = paste0("Histórico 2002–", ANO_HIST_FIM,
                               " + Projeção ", ANO_HIST_FIM + 1L, "–", ano_max(),
                               " com IC 95%"),
             x = "", y = unidade,
             caption = "Fonte: IBGE (histórico); projeção própria.") +
        tema_base
  })

  # ---- Aba 2: VAB Macrossetor -----------------------------------------------
  output$plot_macro <- renderPlot({
      df <- vab_mac() |>
        filter(geo == input$geo, ano <= ano_max()) |>
        mutate(
          tipo      = factor(tipo, levels = c("Histórico", "Projetado")),
          mac_label = coalesce(MACRO_LABEL[macrossetor], macrossetor)
        )

      if (input$macrossetor != "todos")
        df <- df |> filter(macrossetor == input$macrossetor)

      shiny::validate(shiny::need(nrow(df) > 0, "Sem dados para o macrossetor selecionado."))

      if (input$macrossetor == "todos") {
        ggplot(df, aes(x = ano, fill = mac_label, colour = mac_label)) +
          geom_area(data = \(d) filter(d, tipo == "Histórico"),
                    aes(y = vab_nominal / 1e3), position = "stack", alpha = 0.70) +
          geom_area(data = \(d) filter(d, tipo == "Projetado"),
                    aes(y = vab_nominal / 1e3), position = "stack", alpha = 0.35) +
          vl +
          scale_fill_manual(values = PALETA_MACRO, name = "") +
          scale_colour_manual(values = PALETA_MACRO, name = "") +
          scale_x_continuous(breaks = seq(2002, ANO_PROJ_FIM, 3)) +
          scale_y_continuous(labels = label_number(big.mark = ".", decimal.mark = ",",
                                                    accuracy = 1, suffix = " bi", scale = 1)) +
          labs(title    = paste0("VAB por Macrossetor — ", input$geo),
               subtitle = "R$ bilhões (correntes) | empilhado; sombreado = projetado",
               x = "", y = "R$ bilhões",
               caption = "Fonte: IBGE (histórico); projeção própria.") +
          tema_base
      } else {
        ggplot(df, aes(x = ano)) +
          geom_ribbon(aes(ymin = vab_lo95 / 1e3, ymax = vab_hi95 / 1e3),
                      data = \(d) filter(d, tipo == "Projetado", !is.na(vab_lo95)),
                      fill = COR_RIBBN, alpha = 0.20) +
          vl +
          geom_line(aes(y = vab_nominal / 1e3, colour = tipo), linewidth = 1.1) +
          geom_point(aes(y = vab_nominal / 1e3, colour = tipo), size = 2.0) +
          scale_colour_manual(values = c(Histórico = COR_HIST, Projetado = COR_PROJ), name = "") +
          scale_x_continuous(breaks = seq(2002, ANO_PROJ_FIM, 3)) +
          scale_y_continuous(labels = label_number(big.mark = ".", decimal.mark = ",",
                                                    accuracy = 0.1, suffix = " bi")) +
          labs(title    = paste0("VAB — ", MACRO_LABEL[input$macrossetor], " — ", input$geo),
               subtitle = "R$ bilhões (correntes) | IC 95% no período projetado",
               x = "", y = "R$ bilhões",
               caption = "Fonte: IBGE (histórico); projeção própria.") +
          tema_base
      }
  })

  # ---- Aba 3: VAB Atividade -------------------------------------------------
  output$plot_ativ <- renderPlot({
      df <- vab_ativ() |>
        filter(geo == input$geo, ano <= ano_max()) |>
        mutate(
          tipo       = factor(tipo, levels = c("Histórico", "Projetado")),
          ativ_label = coalesce(ATIVIDADE_LABEL[atividade], atividade)
        )

      if (input$atividade != "todas")
        df <- df |> filter(atividade == input$atividade)

      shiny::validate(shiny::need(nrow(df) > 0, "Sem dados para a atividade selecionada."))

      if (input$atividade == "todas") {
        ggplot(df, aes(x = ano, fill = ativ_label, colour = ativ_label)) +
          geom_area(data = \(d) filter(d, tipo == "Histórico"),
                    aes(y = vab_nominal / 1e3), position = "stack", alpha = 0.80) +
          geom_area(data = \(d) filter(d, tipo == "Projetado"),
                    aes(y = vab_nominal / 1e3), position = "stack", alpha = 0.40) +
          vl +
          scale_fill_manual(values = PALETA_ATIV, name = "") +
          scale_colour_manual(values = PALETA_ATIV, name = "") +
          scale_x_continuous(breaks = seq(2002, ANO_PROJ_FIM, 3)) +
          scale_y_continuous(labels = label_number(big.mark = ".", decimal.mark = ",",
                                                    accuracy = 1, suffix = " bi", scale = 1)) +
          labs(title    = paste0("VAB por Atividade — ", input$geo),
               subtitle = "R$ bilhões (correntes) | empilhado; sombreado = projetado",
               x = "", y = "R$ bilhões",
               caption = "Fonte: IBGE (histórico); projeção própria.") +
          tema_base +
          theme(legend.text = element_text(size = 7), legend.key.size = unit(0.4, "cm"))
      } else {
        ggplot(df, aes(x = ano)) +
          geom_ribbon(aes(ymin = vab_lo95 / 1e3, ymax = vab_hi95 / 1e3),
                      data = \(d) filter(d, tipo == "Projetado", !is.na(vab_lo95)),
                      fill = COR_RIBBN, alpha = 0.20) +
          vl +
          geom_line(aes(y = vab_nominal / 1e3, colour = tipo), linewidth = 1.1) +
          geom_point(aes(y = vab_nominal / 1e3, colour = tipo), size = 2.0) +
          scale_colour_manual(values = c(Histórico = COR_HIST, Projetado = COR_PROJ), name = "") +
          scale_x_continuous(breaks = seq(2002, ANO_PROJ_FIM, 3)) +
          scale_y_continuous(labels = label_number(big.mark = ".", decimal.mark = ",",
                                                    accuracy = 0.1, suffix = " bi")) +
          labs(title    = paste0("VAB — ", ATIVIDADE_LABEL[input$atividade], " — ", input$geo),
               subtitle = "R$ bilhões (correntes) | IC 95% no período projetado",
               x = "", y = "R$ bilhões",
               caption = "Fonte: IBGE (histórico); projeção própria.") +
          tema_base
      }
  })

  # ---- Aba 4: Comparativo ---------------------------------------------------
  output$titulo_comp <- renderText({
    paste0("Comparativo — ", VARIAVEL_LABEL[input$variavel])
  })

  output$plot_comp <- renderPlot({
      shiny::validate(shiny::need(length(input$geos_comp) >= 1, "Selecione ao menos um território."))
      df <- serie() |>
        filter(geo %in% input$geos_comp, variavel == input$variavel,
               ano <= ano_max()) |>
        mutate(tipo = factor(tipo, levels = c("Histórico", "Projetado")))

      shiny::validate(shiny::need(nrow(df) > 0, "Sem dados para a combinação selecionada."))

      unidade <- if (eh_indice(input$variavel)) "Índice (2002 = 100)" else "R$ bilhões"

      ggplot(df, aes(x = ano, y = valor, colour = geo, linetype = tipo)) +
        vl +
        camada_base_100(input$variavel) +
        geom_line(linewidth = 0.9) +
        scale_linetype_manual(values = c(Histórico = "solid", Projetado = "dashed"), name = "") +
        scale_colour_brewer(palette = "Set2", name = "") +
        scale_x_continuous(breaks = seq(2002, ANO_PROJ_FIM, 3)) +
        { if (!eh_indice(input$variavel))
            scale_y_continuous(labels = label_number(big.mark = ".", decimal.mark = ",",
                                                      accuracy = 1, scale = 1e-3, suffix = " bi"))
          else
            scale_y_continuous(labels = label_number(suffix = "%", accuracy = 0.1, decimal.mark = ","))
        } +
        labs(title    = paste0(VARIAVEL_LABEL[input$variavel], " — Comparativo"),
             subtitle = paste0("Histórico 2002–", ANO_HIST_FIM,
                               " (contínuo) + Projeção ", ANO_HIST_FIM + 1L, "–", ano_max(),
                               " (tracejado)"),
             x = "", y = unidade,
             caption = "Fonte: IBGE (histórico); projeção própria.") +
        tema_base
  })

  # ---- Aba 5: Tabela --------------------------------------------------------
  output$tabela_dados <- renderDT({
    lbl_valor <- if (eh_indice(input$tabela_var)) "Índice (base 100 = 2002)" else "Valor (R$ milhões)"

    df <- if (input$tabela_fonte == "serie") {
      serie() |>
        filter(geo == input$tabela_geo, variavel == input$tabela_var,
               ano <= ano_max()) |>
        transmute(
          Território = geo, Tipo = tipo, Ano = ano,
          Variável   = coalesce(VARIAVEL_LABEL[variavel], variavel),
          !!lbl_valor := round(valor, 2),
          `Limite inferior (IC 95%)` = round(lo95, 2),
          `Limite superior (IC 95%)` = round(hi95, 2)
        )
    } else if (input$tabela_fonte == "macrossetor") {
      d <- vab_mac() |>
        filter(geo == input$tabela_geo_mac, ano <= ano_max())
      if (input$tabela_mac != "todos")
        d <- d |> filter(macrossetor == input$tabela_mac)
      d |> transmute(
        Território  = geo, Tipo = tipo, Ano = ano,
        Macrossetor = coalesce(MACRO_LABEL[macrossetor], macrossetor),
        `VAB Nominal (R$ milhões)` = round(vab_nominal, 1),
        `Limite inferior (IC 95%)` = round(vab_lo95, 1),
        `Limite superior (IC 95%)` = round(vab_hi95, 1)
      )
    } else {
      d <- vab_ativ() |>
        filter(geo == input$tabela_geo_ativ, ano <= ano_max())
      if (input$tabela_ativ != "todas")
        d <- d |> filter(atividade == input$tabela_ativ)
      d |> transmute(
        Território  = geo, Tipo = tipo, Ano = ano,
        Macrossetor = coalesce(MACRO_LABEL[macrossetor], macrossetor),
        Atividade   = coalesce(ATIVIDADE_LABEL[atividade], atividade),
        `VAB Nominal (R$ milhões)` = round(vab_nominal, 1),
        `Limite inferior (IC 95%)` = round(vab_lo95, 1),
        `Limite superior (IC 95%)` = round(vab_hi95, 1)
      )
    }

    # Formatar colunas numéricas no padrão brasileiro (NA → vazio)
    fmt_br <- function(x, dec = 2)
      ifelse(is.na(x), "",
             formatC(x, format = "f", digits = dec, big.mark = ".", decimal.mark = ",",
                     big.interval = 3L))

    num_cols <- setdiff(names(df)[sapply(df, is.numeric)], "Ano")
    for (col in num_cols)
      df[[col]] <- fmt_br(df[[col]])

    datatable(df,
      options = list(paging = FALSE, scrollX = TRUE, scrollY = "480px",
                     dom = "t",
                     columnDefs = list(list(className = "dt-right", targets = "_all"))),
      rownames = FALSE)
  })

  # ---- Status de dados (footer) -----------------------------------------------
  output$status_dados_txt <- renderUI({
    s <- input$status_dados_raw
    if (is.null(s) || identical(s, "")) {
      return(tags$span(
        paste0("Dados: ", ANO_HIST_INI, "\u2013", ANO_HIST_FIM),
        style = "color:#999; font-size:0.9em;"
      ))
    }
    codigo <- s$codigo_erro
    if (!is.null(codigo) && !is.na(codigo) && nchar(codigo) > 0) {
      tags$span(
        paste0("Aviso: dados desatualizados (", codigo, ")"),
        style = "color:#c0392b; font-weight:600; font-size:0.9em;"
      )
    } else {
      tags$span(
        paste0("Dados: ", s$ano_ini, "\u2013", s$ano_fim,
               " \u00b7 atualizado em ", s$data_download),
        style = "color:#999; font-size:0.9em;"
      )
    }
  })

  # ---- Aba Diagnóstico ---------------------------------------------------------
  output$tabela_diagnostico <- renderDT({
    VARIAVEL_DIAG_LABEL <- c(
      idx_volume   = "Volume",
      idx_preco    = "Preço",
      log_impostos = "Impostos"
    )

    d <- diagnostico_df()

    # Filtros próprios da aba
    if (!is.null(input$diag_geo)     && input$diag_geo     != "todos")
      d <- d |> filter(geo        == input$diag_geo)
    if (!is.null(input$diag_tipo)    && input$diag_tipo    != "todos")
      d <- d |> filter(serie_tipo == input$diag_tipo)
    if (!is.null(input$diag_modelo)  && input$diag_modelo  != "todos")
      d <- d |> filter(modelo     == input$diag_modelo)
    if (isTRUE(input$diag_fallback))
      d <- d |> filter(fallback   == TRUE)

    d <- d |>
      mutate(
        across(c(mase_ponderado, mase_h1, mase_h2, mase_h3), ~ round(.x, 3)),
        variavel    = coalesce(VARIAVEL_DIAG_LABEL[variavel], variavel),
        macrossetor = coalesce(MACRO_LABEL[macrossetor], macrossetor),
        atividade   = coalesce(ATIVIDADE_LABEL[atividade], atividade),
        fallback    = if_else(fallback, "Sim", "\u2014")
      ) |>
      select(
        `Território`  = geo,
        `Tipo`        = serie_tipo,
        `Macrossetor` = macrossetor,
        `Atividade`   = atividade,
        `Variável`    = variavel,
        `Modelo`      = modelo,
        `MASE pond.`  = mase_ponderado,
        `MASE h=1`    = mase_h1,
        `MASE h=2`    = mase_h2,
        `MASE h=3`    = mase_h3,
        `Fallback`    = fallback
      )

    datatable(d,
      rownames = FALSE,
      options  = list(
        paging   = FALSE,
        scrollY  = "500px",
        scrollX  = TRUE,
        dom      = "frti",
        language = list(
          search       = "Buscar:",
          info         = "Mostrando _TOTAL_ s\u00e9ries",
          infoFiltered = "(filtrado de _MAX_ s\u00e9ries)"
        )
      )
    )
  })
}
